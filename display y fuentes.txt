#include <iostream>
#include <Windows.h> //Obtener dimesion ventana
#include <allegro5/allegro.h>
#include <allegro5/allegro_native_dialog.h>
#include <allegro5/allegro_font.h>
#include <allegro5/allegro_ttf.h>
using namespace std;

int main() {

	if (!al_init()) {
		al_show_native_message_box(NULL, "Warning", "Error Critico", "No se pudo crear la ventana", NULL,
			ALLEGRO_MESSAGEBOX_ERROR
		);
		return -1;
	}
	al_init_native_dialog_addon();
	al_init_font_addon();
	al_init_ttf_addon();
	ALLEGRO_FONT* fuente1 = al_load_font("hello honey.otf", 70, NULL);
	ALLEGRO_DISPLAY* ventana = al_create_display(800, 600);
	int ancho = GetSystemMetrics(SM_CXSCREEN);
	int alto = GetSystemMetrics(SM_CYSCREEN);
	if (!ventana) {
		al_show_native_message_box(NULL, "Warning", "Error Critico", "No se pudo crear la ventana", NULL,
			ALLEGRO_MESSAGEBOX_ERROR
		);
		return -1;
	}
	//config
	al_set_window_title(ventana, "MI VENTANA!");
	al_set_window_position(ventana, (ancho)/2 - 800/2, alto/2 - 600/2);
	
	
	while (true) {
		al_draw_text(fuente1, al_map_rgb(255, 255, 255), 100, 100, 0, "Hola Mundo");
		al_flip_display();
		
	}


	return 0;
}
